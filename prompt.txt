    app.config["ENV"] = "development"

    from .helpers import format_cop
    app.jinja_env.filters["cop"] = format_cop

    from .helpers import get_value
    app.jinja_env.filters["get_value"] = get_value

    app.static_folder = "static"
    app.secret_key = os.getenv("FLASK_KEY", "default-secret-key")

    basedir = os.path.abspath(os.path.dirname(__file__))
    project_root = os.path.abspath(os.path.join(basedir, ".."))
    data_dir = os.path.join(project_root, "data")
    db_path = os.path.join(project_root, "database.db")

    app.config["PROJECT_ROOT"] = project_root
    app.config["DATA_DIR"] = data_dir
    app.config["ALL_ORDERS_CSV"] = os.path.join(data_dir, "all_orders.csv")
    app.config["ALL_ORDERS_CACHE_FILE"] = os.path.join(data_dir, ".all_orders_cache_ts")

# app/services/daily_sales.py
import os
import logging
import pandas as pd
from statsmodels.tsa.holtwinters import ExponentialSmoothing
from statsmodels.tsa.statespace.sarimax import SARIMAX


def _forecast_daily_series(series: pd.Series, periods: int = 14) -> pd.Series:
    series = series.astype(float)

    if len(series) < 10:
        last = float(series.iloc[-1]) if len(series) else 0.0
        return pd.Series([last] * periods)

    try:
        seasonal_periods = 7
        use_seasonal = len(series) >= (seasonal_periods * 3)

        model = ExponentialSmoothing(
            series,
            trend="add",
            seasonal="add" if use_seasonal else None,
            seasonal_periods=seasonal_periods if use_seasonal else None,
            initialization_method="estimated",
        )
        fit = model.fit(optimized=True)
        return fit.forecast(periods).clip(lower=0.0)
    except Exception:
        pass

    try:
        seasonal_periods = 7
        use_seasonal = len(series) >= (seasonal_periods * 3)
        seasonal_order = (1, 0, 1, seasonal_periods) if use_seasonal else (0, 0, 0, 0)

        model = SARIMAX(
            series,
            order=(1, 1, 1),
            seasonal_order=seasonal_order,
            enforce_stationarity=False,
            enforce_invertibility=False,
        )
        fit = model.fit(disp=False)
        return fit.forecast(steps=periods).clip(lower=0.0)
    except Exception:
        last = float(series.iloc[-1]) if len(series) else 0.0
        return pd.Series([last] * periods)


def get_daily_sales_trend(
    output_file: str = "daily_sales_trend.csv",
    forecast_periods: int = 14,
    return_forecast: bool = True,
    orders_csv_path: str = "data/all_orders.csv",
    utm_source_filter: str | None = None,
):
    logger = logging.getLogger(__name__)
    tz = "America/Bogota"

    logger.info("Building daily sales trend from %s", orders_csv_path)

    if not os.path.exists(orders_csv_path):
        raise FileNotFoundError(f"Orders data file not found: {orders_csv_path}")

    data = pd.read_csv(orders_csv_path)

    required_cols = {"order_date", "total_value"}
    missing = required_cols - set(data.columns)
    if missing:
        raise ValueError(f"Orders file is missing required columns: {sorted(missing)}")

    data["order_date"] = pd.to_datetime(data["order_date"], errors="coerce")
    before = len(data)
    data = data.dropna(subset=["order_date"])
    after = len(data)
    if after < before:
        logger.warning("Dropped %s rows with invalid order_date", before - after)

    if getattr(data["order_date"].dt, "tz", None) is None:
        data["order_date_local"] = data["order_date"].dt.tz_localize(tz)
    else:
        data["order_date_local"] = data["order_date"].dt.tz_convert(tz)

    data["total_value"] = pd.to_numeric(data["total_value"], errors="coerce").fillna(0.0)

    now_bogota = pd.Timestamp.now(tz=tz).normalize()
    end_yesterday = now_bogota - pd.Timedelta(microseconds=1)
    data = data[data["order_date_local"] <= end_yesterday].copy()
    logger.info("Using orders up to %s (end of yesterday, Bogota)", end_yesterday)

    if utm_source_filter:
        if "utm_source" not in data.columns:
            raise ValueError("Orders file is missing utm_source column (required for utm_source_filter).")
        data["utm_source"] = (
            data["utm_source"]
            .fillna("")
            .astype(str)
            .str.strip()
            .str.lower()
        )
        data = data[data["utm_source"] == utm_source_filter.lower()].copy()
        logger.info("Applied utm_source filter: %s", utm_source_filter)

    output_dir = os.path.join(os.getcwd(), "data")
    os.makedirs(output_dir, exist_ok=True)
    csv_path = os.path.join(output_dir, output_file)

    if data.empty:
        pd.DataFrame(columns=["Date", "Sales"]).to_csv(csv_path, index=False)
        return ([], []) if return_forecast else []

    data["day"] = data["order_date_local"].dt.floor("D")

    total_orders_per_day = (
        data.groupby("day")
        .size()
        .reset_index(name="Total Orders")
    )

    total_sales_per_day = (
        data.groupby("day")["total_value"]
        .sum()
        .reset_index(name="Total_Sales_Num")
    )

    summary = (
        total_orders_per_day
        .merge(total_sales_per_day, on="day", how="left")
        .sort_values("day")
    )

    summary["Total_Sales_Num"] = summary["Total_Sales_Num"].fillna(0.0)
    summary["Date"] = summary["day"].dt.strftime("%Y-%m-%d")
    summary["Total Sales"] = summary["Total_Sales_Num"].apply(lambda x: str(int(round(x))))

    summary_rows = summary[["Date", "Total Orders", "Total Sales"]].to_dict(orient="records")

    csv_summary = summary[["Date", "Total_Sales_Num"]].rename(columns={"Total_Sales_Num": "Sales"})
    csv_summary.to_csv(csv_path, index=False)

    forecast_rows = []
    if return_forecast:
        ts = csv_summary.copy()
        ts["Date_dt"] = pd.to_datetime(ts["Date"], errors="coerce")
        ts = ts.dropna(subset=["Date_dt"]).sort_values("Date_dt")

        if not ts.empty:
            s = ts.set_index("Date_dt")["Sales"].astype(float)

            full_idx = pd.date_range(s.index.min(), s.index.max(), freq="D")
            s = s.reindex(full_idx, fill_value=0.0)

            fc = _forecast_daily_series(s, periods=forecast_periods)
            fc_idx = pd.date_range(s.index.max() + pd.Timedelta(days=1), periods=forecast_periods, freq="D")

            for d, v in zip(fc_idx, fc):
                forecast_rows.append({
                    "Date": d.strftime("%Y-%m-%d"),
                    "Projected Total Sales": float(v),
                })

    return (summary_rows, forecast_rows) if return_forecast else summary_rows


def get_daily_sales_trend_simple(*args, **kwargs):
    return get_daily_sales_trend(*args, **kwargs)

<script>
document.addEventListener("DOMContentLoaded", function() {

  var exchangeRateEl = document.getElementById("exchangeRateDailySales");
  var exchangeRate = parseFloat(exchangeRateEl ? exchangeRateEl.value : "4000");
  var useUSD = false;

  function convertToUSD(values) {
    return values.map(function(v) {
      return Number((v / exchangeRate).toFixed(2));
    });
  }

  function formatYTick(value) {
    return useUSD
      ? "$" + Number(value).toFixed(2) + " USD"
      : new Intl.NumberFormat("es-CO", { style: "currency", currency: "COP" }).format(value);
  }

  function hexToRgba(hex, alpha) {
    if (!hex || hex[0] !== "#" || (hex.length !== 7 && hex.length !== 4)) return null;

    var r, g, b;

    if (hex.length === 7) {
      r = parseInt(hex.slice(1, 3), 16);
      g = parseInt(hex.slice(3, 5), 16);
      b = parseInt(hex.slice(5, 7), 16);
    } else {
      r = parseInt(hex[1] + hex[1], 16);
      g = parseInt(hex[2] + hex[2], 16);
      b = parseInt(hex[3] + hex[3], 16);
    }
    return "rgba(" + r + "," + g + "," + b + "," + alpha + ")";
  }

  var PALETTE = [
    "#4e79a7", "#f28e2b", "#e15759", "#76b7b2", "#59a14f",
    "#edc949", "#af7aa1", "#ff9da7", "#9c755f", "#bab0ab",
    "#1f77b4", "#ff7f0e", "#2ca02c", "#d62728", "#9467bd",
    "#8c564b", "#e377c2", "#7f7f7f", "#bcbd22", "#17becf"
  ];

  var GENDER_COLOR_MAP = {
    "female": "#ff5aa5",
    "male": "#4e79a7",
    "unknown": "#9e9e9e",
    "other": "#af7aa1"
  };

  // Unique color for the "Other" slice in CITY pies (total + channels)
  var CITY_OTHER_COLOR = "#000000";

  function normKey(s) {
    return String(s || "").trim().toLowerCase();
  }

  function buildGenderColors(labels) {
    var out = [];
    for (var i = 0; i < labels.length; i++) {
      var k = normKey(labels[i]);
      out.push(GENDER_COLOR_MAP[k] || "#bab0ab");
    }
    return out;
  }

  // PIE HELPER
  // If useFixedGenderColors is true -> uses GENDER_COLOR_MAP
  // Else -> uses PALETTE (with optional forced colors by label)
  function buildPieChart(canvasId, labels, valuesCOP, titleText, useFixedGenderColors, specialSliceColorMap) {
    var canvas = document.getElementById(canvasId);
    if (!canvas) return null;

    // filter out zeros
    var fl = [];
    var fv = [];
    for (var i = 0; i < labels.length; i++) {
      var v = Number(valuesCOP[i]) || 0;
      if (v > 0) {
        fl.push(labels[i]);
        fv.push(v);
      }
    }

    var total = fv.reduce(function(acc, v) { return acc + (Number(v) || 0); }, 0);

    var colors = [];
    if (useFixedGenderColors) {
      colors = buildGenderColors(fl);
    } else {
      for (var j = 0; j < fl.length; j++) {
        colors.push(PALETTE[j % PALETTE.length]);
      }
    }

    if (specialSliceColorMap) {
      for (var k = 0; k < fl.length; k++) {
        var key = normKey(fl[k]);
        if (specialSliceColorMap[key]) {
          colors[k] = specialSliceColorMap[key];
        }
      }
    }

    var ctx = canvas.getContext("2d");
    var chart = new Chart(ctx, {
      type: "pie",
      data: {
        labels: fl,
        datasets: [{
          data: fv,
          backgroundColor: colors,
          borderColor: colors,
          borderWidth: 1
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          legend: { position: "bottom" },
          title: titleText ? { display: true, text: titleText } : { display: false },
          tooltip: {
            callbacks: {
              label: function(context) {
                var label = context.label || "";
                var value = Number(context.parsed || 0);
                var percent = total > 0 ? (value / total) * 100 : 0;

                var formattedValue = useUSD
                  ? "$" + Number(value).toFixed(2) + " USD"
                  : new Intl.NumberFormat("es-CO", { style: "currency", currency: "COP" }).format(value);

                return label + ": " + formattedValue + " (" + percent.toFixed(1) + "%)";
              }
            }
          },
          datalabels: {
            display: function(ctx) {
              var v = Number(ctx.dataset.data[ctx.dataIndex] || 0);
              return v > 0;
            },
            formatter: function(value, ctx) {
              var sum = ctx.dataset.data.reduce(function(acc, v) { return acc + (Number(v) || 0); }, 0);
              var pct = sum > 0 ? (Number(value) / sum) * 100 : 0;
              return pct.toFixed(1) + "%";
            },
            color: "#fff",
            font: { weight: "bold" }
          }
        }
      },
      plugins: (typeof ChartDataLabels !== "undefined") ? [ChartDataLabels] : []
    });

    chart._rawCOP = fv;
    chart._rawLabels = fl;

    return chart;
  }

  var pieCanvas = document.getElementById("utmSourcePieChartDaily");

  if (pieCanvas) {

        filteredPieValuesCOP.push(val2);
      }
    }

    var pieColors = [];
    for (var j2 = 0; j2 < filteredPieLabels.length; j2++) {
      var color2 = PALETTE[j2 % PALETTE.length];
      pieColors.push(color2);
      channelColorMap[String(filteredPieLabels[j2]).toLowerCase()] = color2;
    }

    var pieTotal = filteredPieValuesCOP.reduce(function(acc, v) {
      return acc + (Number(v) || 0);
    }, 0);

    var pieCtx = pieCanvas.getContext("2d");
    pieChart = new Chart(pieCtx, {
      type: "pie",

        labels: filteredPieLabels,
        datasets: [{
          data: filteredPieValuesCOP,
          backgroundColor: pieColors,
          borderColor: pieColors,
          borderWidth: 1
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          legend: { position: "bottom" },
          tooltip: {
            callbacks: {
              label: function(context) {
                var label = context.label || "";
                var value = Number(context.parsed || 0);
                var percent = pieTotal > 0 ? (value / pieTotal) * 100 : 0;

                var formattedValue = useUSD
                  ? "$" + Number(value).toFixed(2) + " USD"
                  : new Intl.NumberFormat("es-CO", { style: "currency", currency: "COP" }).format(value);

                return label + ": " + formattedValue + " (" + percent.toFixed(1) + "%)";
              }
            }
          },
          datalabels: {
            formatter: function(value) {
              var pct = pieTotal > 0 ? (Number(value) / pieTotal) * 100 : 0;
              return pct.toFixed(1) + "%";
            },
            color: "#fff",
            font: { weight: "bold" }
          }
        }
      },
      plugins: (typeof ChartDataLabels !== "undefined") ? [ChartDataLabels] : []
    });

    pieChart._rawCOP = filteredPieValuesCOP;
  }

  // ======================
  // TOTAL PIES
  // ======================
  var genderLabelsTotal = [
    {% for l in gender_labels_total %}
      "{{ l }}",
    {% endfor %}
  ];
  var genderValuesTotal = [
    {% for v in gender_values_total %}
      {{ v }},
    {% endfor %}
  ];
  var genderPieTotal = buildPieChart("genderPieTotal", genderLabelsTotal, genderValuesTotal, "", true);

  var cityLabelsTotal = [
    {% for l in city_labels_total %}
      "{{ l }}",
    {% endfor %}
  ];
  var cityValuesTotal = [
    {% for v in city_values_total %}
      {{ v }},
    {% endfor %}
  ];
  var cityPieTotal = buildPieChart(
    "cityPieTotal",
    cityLabelsTotal,
    cityValuesTotal,
    "",
    false,
    { "other": CITY_OTHER_COLOR }
  );

  var timeLabelsTotal = [
    {% for l in time_labels_total %}
      "{{ l }}",
    {% endfor %}
  ];
  var timeValuesTotal = [
    {% for v in time_values_total %}
      {{ v }},
    {% endfor %}
  ];
  var timePieTotal = buildPieChart("timePieTotal", timeLabelsTotal, timeValuesTotal, "", false);

  // ======================
  // LINE CHART FACTORY
  // ======================
  function buildLineChart(canvasId, labels, histCOP, fcstCOP, titlePrefix, hexColor) {
    var canvas = document.getElementById(canvasId);
    if (!canvas) return null;

    var ctx = canvas.getContext("2d");

    var fillColor = hexToRgba(hexColor, 0.15) || undefined;
    var lineColor = hexColor || undefined;

    var chart = new Chart(ctx, {
      type: "line",
      data: {
        labels: labels,
        datasets: [
          {
            label: titlePrefix + " (COP)",
            data: histCOP,
            fill: true,
            borderColor: lineColor,
            backgroundColor: fillColor
          },
          {
            label: "Forecasted " + titlePrefix + " (COP)",
            data: [].concat(new Array(histCOP.length).fill(null), fcstCOP),
            borderDash: [5, 5],
            fill: true,
            borderColor: lineColor,
            backgroundColor: fillColor
          }
        ]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: { legend: { position: "top" } },
        scales: { y: { ticks: { callback: formatYTick } } }
      }
    });

    chart._rawHistCOP = histCOP;
    chart._rawForecastCOP = fcstCOP;
    chart._histLen = histCOP.length;
    chart._titlePrefix = titlePrefix;
    chart._hexColor = hexColor;

    return chart;
  }

  // ======================
  // CURRENCY TOGGLE
  // ======================
  window.toggleCurrencyDailySales = function() {
    exchangeRate = parseFloat(exchangeRateEl.value || "4000");
    useUSD = !useUSD;

    function updatePie(pie) {
      if (!pie || !pie._rawCOP) return;
      pie.data.datasets[0].data = useUSD ? convertToUSD(pie._rawCOP) : pie._rawCOP;
      pie.update();
    }

    updatePie(pieChart);

    updatePie(genderPieTotal);
    updatePie(cityPieTotal);
    updatePie(timePieTotal);

    for (var p = 0; p < channelGenderPies.length; p++) updatePie(channelGenderPies[p]);
    for (var q = 0; q < channelCityPies.length; q++) updatePie(channelCityPies[q]);
    for (var r = 0; r < channelTimePies.length; r++) updatePie(channelTimePies[r]);

    function applyLine(chart) {
      if (!chart) return;

      var hist = useUSD ? convertToUSD(chart._rawHistCOP) : chart._rawHistCOP;
      var fcst = useUSD ? convertToUSD(chart._rawForecastCOP) : chart._rawForecastCOP;

      chart.data.datasets[0].data = hist;
      chart.data.datasets[1].data = [].concat(new Array(chart._histLen).fill(null), fcst);

      chart.data.datasets[0].label = useUSD ? chart._titlePrefix + " (USD)" : chart._titlePrefix + " (COP)";
      chart.data.datasets[1].label = useUSD
        ? "Forecasted " + chart._titlePrefix.replace("Historical ", "") + " (USD)"
        : "Forecasted " + chart._titlePrefix.replace("Historical ", "") + " (COP)";

      chart.update();
    }

    applyLine(dailySalesChart);
    for (var i = 0; i < channelCharts.length; i++) applyLine(channelCharts[i]);
  };

});
